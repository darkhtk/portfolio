export class CoordinatorAgent extends AgentBase {
  readonly role: AgentRole = 'coordinator';

  // 1. 어떤 이벤트에 깨어날지 선언
  readonly wakeOn = [
    'file:board-changed',
    'task:moved-to-review',
    'task:approved',
    'task:rejected',
    'clock:tick',
  ];

  // 2. 깨어날지 결정 (board 상태 + 이벤트 type 기반)
  shouldWake(event: OrchestratorEvent, board: Board): boolean {
    if (board.frozen) return false;
    if (event.type === 'task:approved') return true;
    return board.backlog.length < 3;
  }

  // 3. 시스템 prompt + 동적 컨텍스트로 prompt 빌드
  async buildPrompt(event, board): Promise<string> {
    const systemPrompt = await this.loadSystemPrompt();
    const sections = [systemPrompt];
    sections.push(`Rejected: ${board.rejected.length}`);
    sections.push(`In Progress: ${board.inProgress.length}`);
    return sections.join('\n');
  }

  // 4. structured JSON output schema
  getOutputSchema(): object {
    return COORDINATOR_OUTPUT_SCHEMA;
  }
}

전체 GitHub → 
export class CoordinatorAgent extends AgentBase {
  readonly role: AgentRole = 'coordinator';
  readonly wakeOn = [
    'file:board-changed',
    'task:moved-to-review',
    'task:approved',
    'task:rejected',
    'clock:tick',
  ];

  private lastSyncAt?: Date;

  shouldWake(event: OrchestratorEvent, board: Board): boolean {
    if (board.frozen) return false;

    // clock:tick 은 5분 이상 간격 보장
    if (event.type === 'clock:tick') {
      if (this.lastSyncAt) {
        const elapsed = Date.now() - this.lastSyncAt.getTime();
        if (elapsed < 300_000) return false;
      }
      return true;
    }

    // review 관련 이벤트 → 즉시 wake
    if (event.type === 'task:approved' ||
        event.type === 'task:rejected') {
      return true;
    }

    return board.backlog.length < 3;
  }
}

export class DeveloperAgent extends AgentBase {
  readonly role: AgentRole = 'developer';
  readonly wakeOn = [
    'task:status-changed',
    'task:approved',
    'task:rejected',
    'task:created',
    'clock:tick',
  ];

  shouldWake(event: OrchestratorEvent, board: Board): boolean {
    if (board.frozen) return false;

    const hasRejected = board.rejected.length > 0;
    const hasInProgress = board.inProgress.some(
      (t) => t.assignee === 'developer'
    );
    const hasBacklog = board.backlog.length > 0;

    // rejected → 무조건 wake (수정 우선)
    if (event.type === 'task:rejected') return true;

    // approved → 다음 픽업할 게 있으면 wake
    if (event.type === 'task:approved') return hasBacklog;

    return hasRejected || hasInProgress || hasBacklog;
  }
}

export class SupervisorAgent extends AgentBase {
  readonly role: AgentRole = 'supervisor';
  readonly wakeOn = ['task:done', 'clock:tick', 'agent:errored'];

  private auditCycleIndex = 0;
  private readonly AUDIT_CYCLE = [
    'asset_prep',
    'code_audit',
    'performance',
    'ux_improvement',
    'error_check',
  ] as const;

  async buildPrompt(event, board): Promise<string> {
    const systemPrompt = await this.loadSystemPrompt();
    const sections = [systemPrompt];

    // 매 루프 다음 audit action으로 자동 순환
    const action = this.AUDIT_CYCLE[
      this.auditCycleIndex % this.AUDIT_CYCLE.length
    ];
    this.auditCycleIndex++;

    sections.push(`## Current Action: ${action}`);
    switch (action) {
      case 'asset_prep':
        sections.push('🎨 RESERVE 다음 5건 에셋 미리 생성');
        break;
      case 'code_audit':
        sections.push('주요 코드 파일 읽고 버그 직접 수정');
        break;
      // ... performance, ux, error
    }
    return sections.join('\n');
  }
}

export class ClientAgent extends AgentBase {
  readonly role: AgentRole = 'client';
  readonly wakeOn = [
    'task:moved-to-review',
    'task:created',
    'clock:tick'
  ];

  shouldWake(_event, board): boolean {
    if (board.frozen) return false;
    // verdict 없는 In Review 태스크가 있으면 wake
    return board.inReview.some((t) => !t.reviewVerdict);
  }

  async buildPrompt(_event, board): Promise<string> {
    const systemPrompt = await this.loadSystemPrompt();
    const sections = [systemPrompt];

    const pending = board.inReview.find((t) => !t.reviewVerdict);
    if (!pending) return sections.join('\n');

    sections.push(`## Review Task: ${pending.id} — ${pending.title}`);
    if (pending.specRef) {
      sections.push(`Spec: orchestration/specs/${pending.specRef}.md`);
    }

    sections.push('1. Read the spec and the implementation code');
    sections.push('2. Perform all 4 verifications');
